##### Simple

<Block>
  <Breadcrumbs
    links={[
      { name: 'Home', href: '/' },
      { name: 'Components', href: '/components' },
      { name: 'Breadcrumbs' },
    ]}
  />
</Block>

```jsx
import { Breadcrumbs } from '../../src/components';

// ----------------------------------------------------------------------

<Breadcrumbs
  links={[
    { name: 'Home', href: '/' },
    { name: 'Components', href: '/components' },
    { name: 'Breadcrumbs' },
  ]}
/>;
```

```jsx
import { Breadcrumbs, Iconify } from '../../src/components';

      icon: <Iconify icon="carbon:grid" />,

      name: 'Breadcrumbs',

```
